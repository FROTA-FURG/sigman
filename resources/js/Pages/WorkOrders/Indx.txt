import SIGMANLayout from '@/Layouts/SIGMANLayout';
import { Head } from '@inertiajs/react';
import { useState, useMemo } from 'react';
import { pdf } from '@react-pdf/renderer';
import OSPdfTemplate from './components/OSPdfTemplate';
import CreateWorkOrderModal from '@/Pages/WorkOrders/components/CreateOSModal';
import EditWorkOrderModal from './components/EditWorkOrderModal';
import WorkOrderDetailsModal from './components/WorkOrderDetailsModal';

export default function Index({ workOrders = [], equipments = [], users = []}) {
    const [isModalOpen, setIsModalOpen] = useState(false);
    const [isDetailsModalOpen, setIsDetailsModalOpen] = useState(false);
    const [selectedOsId, setSelectedOsId] = useState(null);

    const [isEditModalOpen, setIsEditModalOpen] = useState(false);
    const [osToEdit, setOsToEdit] = useState(null);

    const [vesselFilter, setVesselFilter] = useState('');
    const [statusFilter, setStatusFilter] = useState('');

    const [sortConfig, setSortConfig] = useState({ key: 'created_at', direction: 'desc' });

    const statusLabels = {
        open: 'Aberto',
        in_progress: 'Em Andamento',
        completed: 'Fechado',
        cancelled: 'Cancelado',
    };

    const selectedOs = workOrders.find(os => os.id === selectedOsId);
    const workOrderActivities = selectedOs?.activities || [];

    const handleOpenModal = () => setIsModalOpen(true);

    const handleOpenDetailsModal = (id) => {
        setSelectedOsId(id);
        setIsDetailsModalOpen(true);
    };

    const handleRowClick = (os) => {
        setOsToEdit(os);
        setIsEditModalOpen(true);
    };

    const handleSort = (key) => {
        let direction = 'asc';
        if (sortConfig.key === key && sortConfig.direction === 'asc') {
            direction = 'desc';
        }
        setSortConfig({ key, direction });
    };

    const sortedAndFilteredWorkOrders = useMemo(() => {
        let result = workOrders;
        if (vesselFilter) {
            result = result.filter(os => os.equipment?.vessel?.tag === vesselFilter);
        }

        if (statusFilter){
            result = result.filter(os => os.status === statusFilter);
        }

        // Aplica a ordenação
        if (sortConfig.key) {
            result = [...result].sort((a, b) => {
                const aValue = a[sortConfig.key] || '';
                const bValue = b[sortConfig.key] || '';

                if (aValue < bValue) {
                    return sortConfig.direction === 'asc' ? -1 : 1;
                }
                if (aValue > bValue) {
                    return sortConfig.direction === 'asc' ? 1 : -1;
                }
                return 0;
            });
        }
        return result;
    }, [workOrders, vesselFilter, statusFilter, sortConfig]);

    const uniqueVessels = useMemo(() => {
        const tags = workOrders.map(os => os.equipment?.vessel?.tag).filter(Boolean);
        return [...new Set(tags)].sort();
    }, [workOrders]);

    const uniqueStatus = useMemo(() => {
        const status = workOrders
            .map(os => os.status)
            .filter(Boolean);

        return [...new Set(status)].sort();
    }, [workOrders]);

    const filteredWorkOrders = useMemo(() => {
        if (!vesselFilter) return workOrders;
        return workOrders.filter(os => os.equipment?.vessel?.tag === vesselFilter);

        switch (status) {
            case 'completed': return <span className="inline-flex items-center rounded-md bg-green-500/10 px-2 py-1 text-xs font-medium text-green-400 ring-1 ring-inset ring-green-500/20">Fechado</span>;
            case 'open': return <span className="inline-flex items-center rounded-md bg-blue-500/10 px-2 py-1 text-xs font-medium text-blue-400 ring-1 ring-inset ring-blue-500/20">Aberto</span>;
            case 'in_progress': return <span className="inline-flex items-center rounded-md bg-yellow-500/10 px-2 py-1 text-xs font-medium text-yellow-400 ring-1 ring-inset ring-yellow-500/20">Em Andamento</span>;
            case 'cancelled': return <span className="inline-flex items-center rounded-md bg-red-500/10 px-2 py-1 text-xs font-medium text-red-400 ring-1 ring-inset ring-red-500/20">Cancelado</span>;
            default: return <span className="inline-flex items-center rounded-md bg-slate-500/10 px-2 py-1 text-xs font-medium text-slate-400 ring-1 ring-inset ring-slate-500/20">{status}</span>;
        }

        switch (priority) {
            case 'critical': return <span className="text-red-400 font-bold">Crítica</span>;
            case 'high': return <span className="text-orange-400 font-medium">Alta</span>;
            case 'medium': return <span className="text-yellow-400">Média</span>;
            case 'low': return <span className="text-slate-400">Baixa</span>;
            default: return priority;
        }
    };

        if (sortConfig.key !== columnKey) {
            // Se não estiver ordenando por esta coluna, mostra um ícone apagado no hover
            return (
                <svg className="w-4 h-4 opacity-0 group-hover:opacity-30 transition-opacity ml-1" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M7 16V4m0 0L3 8m4-4l4 4m6 0v12m0 0l4-4m-4 4l-4-4" />
                </svg>
            );
        }
        // Se estiver ordenando, mostra a seta para cima (asc) ou para baixo (desc) azul
        return sortConfig.direction === 'asc' ? (
            <svg className="w-4 h-4 ml-1 text-blue-400" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M5 15l7-7 7 7" />
            </svg>
        ) : (
            <svg className="w-4 h-4 ml-1 text-blue-400" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M19 9l-7 7-7-7" />
            </svg>
        );
    };

        // Gera o blob do PDF invisivelmente
        const blob = await pdf(<OSPdfTemplate os={os} />).toBlob();
        
        // Cria um link temporário e simula o clique de download
        const url = URL.createObjectURL(blob);
        const link = document.createElement('a');
        link.href = url;
        link.download = `SIGMAN_OS_${os.os_number || 'Sem_Numero'}.pdf`;
        document.body.appendChild(link);
        link.click();
        
        document.body.removeChild(link);
        URL.revokeObjectURL(url);
    };

                            >
                                <option value="">Todas as Embarcações</option>

                                            {vesselFilter

                                                <td className="px-4 py-3 font-bold text-white">{os.ss_number || '-'}</td>
                                                <td className="px-4 py-3 text-slate-400 text-xs">
                                                    {os.created_at ? os.created_at.substring(0, 10).split('-').reverse().join('/') : '-'}
                                                </td>
                                                <td className="px-4 py-3 font-bold text-white">{vessel.tag || '-'}</td>
                                                <td className="px-4 py-3 text-slate-300 font-medium whitespace-normal max-w-[200px]">{equip.name || 'Equipamento Removido'}</td>
                                                <td className="px-4 py-3 text-blue-400 font-mono text-xs">{equip.tag_number || '-'}</td>
                                                
                                                <td className="px-4 py-3 text-slate-300 whitespace-normal max-w-[300px]">

                                                </td>
                                                
                                                <td className="px-4 py-3 text-slate-300 text-xs capitalize">
                                                    {os.maintenance_type === 'corrective' ? 'Corretiva' : os.maintenance_type === 'preventive' ? 'Preventiva' : 'Preditiva'}
                                                </td>
                                                <td className="px-4 py-3 text-xs">{renderPriorityBadge(os.priority)}</td>
                                                <td className="px-4 py-3 text-center">{renderStatusBadge(os.status)}</td>
